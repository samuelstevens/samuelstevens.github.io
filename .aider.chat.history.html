#!/usr/bin/env python3

import sys
import json
import time
import requests
from bs4 import BeautifulSoup
from pathlib import Path

def get_title(url):
    try:
        response = requests.get(url)
        response.raise_for_status()
        soup = BeautifulSoup(response.text, 'html.parser')
        title = soup.title.string.strip() if soup.title else url
        return title
    except Exception as e:
        print(f"Warning: Could not fetch title ({e}), using URL as title")
        return url

def main():
    if len(sys.argv) != 2:
        print("Usage: add_link.py <url>")
        sys.exit(1)

    url = sys.argv[1]
    json_path = Path(__file__).parent.parent / "misc" / "links.json"
    
    # Read existing JSON
    with open(json_path, 'r') as f:
        data = json.load(f)
    
    # Create new link entry
    new_link = {
        "title": get_title(url),
        "link": url,
        "added": int(time.time() * 1000)  # Convert to milliseconds
    }
    
    # Add to links array
    data["links"].append(new_link)
    
    # Write back to file
    with open(json_path, 'w') as f:
        json.dump(data, f, indent=2)
    
    print(f"Added: {new_link['title']}")

if __name__ == "__main__":
    main()

def normalize_arxiv_url(url):
    """Convert arxiv URLs to abs form for title fetching"""
    parsed = urlparse(url)
    if "arxiv.org" not in parsed.netloc:
        return url, url
    
    # Extract the ID from the path
    match = re.search(r'/(pdf|abs)/(\d+\.\d+)', parsed.path)
    if not match:
        return url, url
        
    arxiv_id = match.group(2)
    abs_url = f"https://arxiv.org/abs/{arxiv_id}"
    pdf_url = f"https://arxiv.org/pdf/{arxiv_id}"
    
    # Return the URL to fetch title from, and URL to store
    return abs_url, pdf_url if "pdf" in url else abs_url
